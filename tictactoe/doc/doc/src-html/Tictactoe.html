001import java.util.*;

003/**
004 * A scalable Tic tac toe game. 
005 *
006 * One player tictactoe game versus an AI. This version of the game is played in command prompt. 
007 *
008 * @author Samu Willman
009 * @version 1.0
010 */
011public class Tictactoe {
012    
013    /**
014     * User input to declare the size of the gameboard.
015     * 
016     */
017    static int size = 0;
018    /**
019     * A scanner which is needed in order to take user input.
020     *
021     */
022    static Scanner input = new Scanner(System.in);
023    /**
024     * 2D array and the gameboard.
025     *
026     */
027    static char [][] gameboard;
028    /**
029     * A character which player uses, computer will use O.
030     *
031     */
032    static char playerChar = 'X';
033    /**
034     * Indicates the row in which user will place his marker.
035     *
036     */
037    static int row = 0;
038    /**
039     * Indicates the column in which user will place his marker.
040     *
041     */
042    static int column = 0;
043    /**
044     * Indicates the row in which computer will place his marker.
045     *
046     */
047    static int computerplacerows = 0;
048    /**
049     * Indicates the column in which computer will place his marker.
050     *
051     */
052    static int computerplacecolumns = 0;
053    /**
054     * isChar is shorter version of isThereChar();
055     *
056     */
057    static boolean isChar;
058    /**
059     * This value is used in wincheckrows(), winCheckcolumns and winCheckdiagonal to help with checking 3 or more X's in a row
060     *
061     */
062    static int winConditionX = 0;
063    /**
064     * This value is used in wincheckrows(), winCheckcolumns and winCheckdiagonal to help with checking 3 or more O's in a row
065     *
066     */
067    static int winConditionO = 0;
068    /**
069     * Variable that user gives. This value changes how many X's or O's there has to be in a row
070     *
071     */
072    static int neededToWin = 0;
073    /**
074     * This variable counts rounds, if enough rounds have been played it will be a tie.
075     *
076     */
077    static int roundcounter = 0;
078    /**
079     * Acts as a counter for diagonal win method.
080     *
081     */
082    static int counter = 0;
083    /**
084     * Main method. Initializes the game and has a while loop which keeps the game running until someone wins or it's a tie.
085     * @param args Command line parameters.
086     */
087    public static void main(String[] args) {
088        setBoard();
089        printBoard();
090        System.out.println();
091     
092        while (true) {
093            System.out.println("Player X turn");
094            System.out.println();
095            playerPlace();
096            printBoard(); 
097            if (winCheckrows() == true || winCheckcolumns() == true || winCheckdiagonal() == true) {
098            System.out.println("X WON!!");
099            System.exit(1);
100            }
101            roundcounter++;
102            isBoardFull();
103            turn();
104            System.out.println();
105
106            System.out.println("Computer turn");
107            System.out.println();
108            computerMove();
109            printBoard();
110            if (winCheckrows() == true || winCheckcolumns() == true || winCheckdiagonal() == true) {
111            System.out.println("O WON!!");
112            System.exit(1);
113            }
114            roundcounter++;
115            isBoardFull();
116            turn();
117            System.out.println();
118        }
119    }
120
121    /**
122     * Asks user the size of the gameboard and initializes the gameboard.
123     *
124     */
125    public static void setBoard() {
126        do {
127            System.out.println("Please give board size (min 3x3, max 8x8) (Please note: Give it in the form of a single number!)");
128            try { 
129                size = Integer.parseInt(input.nextLine().trim());
130            } catch (NumberFormatException e) {
131                System.out.println("I only accept numbers!");
132                continue;
133            }
134        } while (size < 3 || size > 8);
135            do {
136                System.out.println("How many marks in a row are required in order to win the game?");
137                try { 
138                    neededToWin = Integer.parseInt(input.nextLine().trim());
139                } catch (NumberFormatException e) {
140                    System.out.println("I only accept numbers!");
141                    continue;
142                }
143            } while (neededToWin < 3 || neededToWin > size);
144        System.out.println();
145        gameboard = new char [size][size];
146    }
147    
148    /**
149     * Prints gameboard and creates coordinate numbers which makes placing marks easier.
150     *
151     */
152    public static void printBoard() {
153            System.out.print("   ");
154            for(int i=0; i<gameboard.length; i++) {
155                System.out.print(i+1 + " ");
156            }
157            System.out.println();
158            for (int i = 0; i < gameboard.length; i++) {
159                if(i < 9) {
160                    System.out.print(" " + (i+1) + " ");
161                } else {
162                    System.out.print(i+1 + " ");
163                }
164                for (int j = 0;j < gameboard[i].length; j++) {
165                    if(gameboard[i][j] != 'X' && gameboard[i][j] != 'O') {
166                        System.out.print("- ");
167                    } else {
168                        System.out.print(gameboard[i][j] + " ");
169                    }
170                }
171                System.out.println();
172            }
173    }
174    
175    /**
176     * Asks the user two numbers which will be used for placing a mark. 
177     * Method also leaves the mark after checking there is no other character in the way.
178     */
179    public static void playerPlace() {
180        isChar = true;
181        while(isChar) {
182            do {
183                System.out.println("Please give a row where you want to place. (Example: 3)");
184                try { 
185                    row = Integer.parseInt(input.nextLine().trim());                         
186                } catch (NumberFormatException e) {
187                    System.out.println("I only accept numbers!");
188                }
189            } while (row < 1 || row > gameboard[0].length);
190            do {
191                System.out.println("Please give a column where you want to place. (Example: 3)");
192                try { 
193                    column = Integer.parseInt(input.nextLine().trim());                         
194                } catch (NumberFormatException e) {
195                    System.out.println("I only accept numbers!");
196                }
197            } while (column < 1 || column > gameboard[0].length);
198            isChar = isThereChar();
199            if(isChar == false) {
200                gameboard[row-1][column-1] = playerChar;
201                break;
202            }
203        }
204        System.out.println();
205    } 
206    
207    /**
208     * This method is used to check if either user or computer has already placed a marker on the wanted position.
209     * @return if there is a character or not
210     */
211    public static boolean isThereChar() {
212        if(gameboard[row-1][column-1] != 'X' && gameboard[row-1][column-1] != 'O') {
213            return false;
214        } else {
215            if(playerChar == 'X') {
216                System.out.println("That spot is occupied already");
217            }
218            return true; 
219        }
220    }
221
222    /**
223     * Computer takes a random number between 1 and the size of the gameboard. 
224     * After picking a number it checks if there is a character in its way before it can place its own character,
225     */
226    public static void computerMove() {
227        isChar = true;
228        while(isChar) {
229            row = (int) (Math.random() * gameboard.length) + 1;
230            column = (int) (Math.random() * gameboard.length) + 1;
231            isChar = isThereChar();
232            if(isChar == false) {
233                gameboard[row-1][column-1] = playerChar;
234                break;
235            }
236        }
237    }
238    
239    /**
240     *Switches the user's and computer's character back and forth which is used for altering between user's and computer's turns.
241     *
242     */
243    public static void turn() {
244        if (playerChar == 'X') {
245            playerChar = 'O';
246            } else {
247            playerChar = 'X';
248            }
249        }
250    
251    /**
252     * Checks gameboard for vertical wins.
253     * @return true if the game is won
254    */
255    public static boolean winCheckrows()  {
256        for(int i = 0; i < gameboard.length; i++){
257            winConditionO = 0;
258            winConditionX = 0;
259            for(int j = 0; j < gameboard[i].length; j++){
260                if(gameboard[i][j] == 'X') {
261                    winConditionX++;
262            }  else {
263                winConditionX = 0;
264            }
265            if (gameboard[i][j] == 'O') {
266                winConditionO++;
267            } else {
268                winConditionO = 0;
269            }
270            if (winConditionO == neededToWin) {
271                return true;
272            }
273            if (winConditionX == neededToWin) {
274               return true;
275        }
276    }
277}
278return false;
279}
280
281    /**
282     * Checks gameboard for horizontal wins.
283     * @return true if the game is won
284    */
285    public static boolean winCheckcolumns()  {
286        for(int i = 0; i < gameboard.length; i++){
287            winConditionO = 0;
288            winConditionX = 0;
289            for(int j = 0; j < gameboard[i].length; j++){
290                if(gameboard[j][i] == 'X') {
291                    winConditionX++;
292            }  else {
293                winConditionX = 0;
294            }
295            if (gameboard[j][i] == 'O') {
296                winConditionO++;
297            } else {
298                winConditionO = 0;
299            }
300            if (winConditionO == neededToWin) {
301                return true;
302            }
303            if (winConditionX == neededToWin) {
304               return true;
305        }
306    }
307}
308return false;
309}

311    /**
312     * Checks gameboard for diagonal wins.
313     * @return true If the game is won
314    */
315public static boolean winCheckdiagonal()  { 
316    for(int i = 0; i < gameboard.length; i++){
317        for(int j = 0; j < gameboard[i].length; j++){
318                if(gameboard[i][j] == playerChar) {
319                counter=1;
320                for(int a=1; a <=neededToWin && i+a<gameboard.length && j + a< gameboard[i].length;a++) { 
321                    if(playerChar == gameboard[i+a][j+a]) {
322                        counter++;
323                        if(counter >= neededToWin) {
324                            return true;
325                        }
326                    } else {
327                        break;
328                    }
329                }
330                counter = 1;
331                for(int a = 1; a<=neededToWin && i+a <gameboard.length && j-a>=0; a++) {
332                    if(playerChar == gameboard[i+a][j-a]) {
333                    counter++;
334                    if(counter >= neededToWin) {
335                        return true;
336                    }
337                } else {
338                    break;
339                }
340            }
341        }
342    }
343
344}
345return false;
346}

348    /**         
349     * Checks gameboard for a draw.
350     * 
351    */
352    public static void isBoardFull () {
353        if(roundcounter == (Math.pow(size, 2) - 1)) {
354            System.out.println("It's a tie!");
355            System.exit(1);
356        }
357    }
